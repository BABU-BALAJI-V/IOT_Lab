{

    pinMode(IRpin,INPUT);

}
void loop()
{
    bool IRstatus=digitalRead(IRpin);
    if(IRstatus==true)
    {
        digitalWrite(Ledpin,LOW);
        Serial.print(IRstatus);
    }
    else
    {
        digitalWriite(Ledpin,HIGH);
        Serial.print(IRstatus);
    }
}
